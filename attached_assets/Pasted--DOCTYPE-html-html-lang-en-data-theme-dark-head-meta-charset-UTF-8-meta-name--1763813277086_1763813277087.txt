  <script>
    // Fetch Infowars Hourly Videos from RSS
    async function fetchInfowarsVideos() {
      const rssUrl = "https://rss.infowars.com/hourly/InfowarsHourlyVideo.xml";
      try {
        const response = await fetch(`https://api.allorigins.win/get?url=${encodeURIComponent(rssUrl)}`);
        const data = await response.json();
        if (!data.contents) throw new Error("No data received");
        const xml = data.contents;
        const parser = new DOMParser();
        const doc = parser.parseFromString(xml, "text/xml");
        const items = doc.querySelectorAll("item");
        const videos = [];
        items.forEach(item => {
          const title = item.querySelector("title")?.textContent;
          const link = item.querySelector("link")?.textContent;
          const description = item.querySelector("description")?.textContent;
          const pubDate = item.querySelector("pubDate")?.textContent;
          const thumbnail = item.querySelector("media\\:thumbnail")?.getAttribute("url");
          const videoId = extractVideoId(link);
          if (videoId && title) {
            videos.push({
              videoId,
              title,
              description,
              pubDate,
              thumbnail: thumbnail || `https://i.ytimg.com/vi/${videoId}/hqdefault.jpg`
            });
          }
        });
        return videos;
      } catch (e) {
        console.error("Fetch error:", e);
        return [];
      }
    }

    // Extract YouTube video ID from URL
    function extractVideoId(url) {
      const regExp = /^.*(youtu.be\/|v\/|u\/\w\/|embed\/|watch\?v=|&v=)([^#&?]*).*/;
      const match = url.match(regExp);
      return (match && match[2].length === 11) ? match[2] : null;
    }

    // Load playlist
    async function loadPlaylist() {
      const container = document.getElementById('playlist-items');
      container.innerHTML = '<div class="loading">Loading Infowars hourly videos…</div>';
      try {
        const videos = await fetchInfowarsVideos();
        if (videos.length === 0) {
          container.innerHTML = '<div class="error">No videos found. <button onclick="loadPlaylist()">Retry</button></div>';
          return;
        }
        container.innerHTML = '';
        videos.forEach(video => {
          const div = document.createElement('div');
          div.className = 'playlist-item';
          div.innerHTML = `
            <img src="${video.thumbnail}" class="thumbnail" loading="lazy">
            <div class="info">
              <h3>${video.title}</h3>
              <div class="channel">${new Date(video.pubDate).toLocaleDateString()}</div>
            </div>
          `;
          div.onclick = () => playVideo(video.videoId, div);
          container.appendChild(div);
        });
        // Auto-play the first video
        if (videos.length > 0) {
          playVideo(videos[0].videoId, container.firstChild);
        }
      } catch (e) {
        container.innerHTML = '<div class="error">Failed to load videos. <button onclick="loadPlaylist()">Retry</button></div>';
      }
    }

    // Play video
    function playVideo(videoId, element) {
      const player = document.getElementById("heroPlayer");
      player.src = `https://www.youtube.com/embed/${videoId}?autoplay=1&rel=0`;
      document.querySelectorAll('.playlist-item').forEach(i => i.classList.remove('selected'));
      element.classList.add('selected');
    }

    // Auto-load on hover
    document.getElementById('playlist-container').addEventListener('mouseenter', () => {
      if (!window.playlistLoaded) {
        loadPlaylist();
        window.playlistLoaded = true;
      }
    });

    // Initialize
    loadPlaylist();
  </script>
